136void HTTPServer::handle_not_found(AsyncWebServerRequest* request) {
137 debugD("NOT_FOUND: ");
138 if (request->method() == HTTP_GET) {
139 debugD("GET");
140 } else if (request->method() == HTTP_POST) {
141 debugD("POST");
142 } else if (request->method() == HTTP_DELETE) {
143 debugD("DELETE");
144 } else if (request->method() == HTTP_PUT) {
145 debugD("PUT");
146 } else if (request->method() == HTTP_PATCH) {
147 debugD("PATCH");
148 } else if (request->method() == HTTP_HEAD) {
149 debugD("HEAD");
150 } else if (request->method() == HTTP_OPTIONS) {
151 debugD("OPTIONS");
152 } else {
153 debugD("UNKNOWN");
154 }
155 debugD(" http://%s%s\n", request->host().c_str(), request->url().c_str());
156
157 if (request->contentLength()) {
158 debugD("_CONTENT_TYPE: %s\n", request->contentType().c_str());
159 debugD("_CONTENT_LENGTH: %u\n", request->contentLength());
160 }
161
162 int headers = request->headers();
163 for (int i = 0; i < headers; i++) {
164 AsyncWebHeader* h = request->getHeader(i);
165 debugD("_HEADER[%s]: %s\n", h->name().c_str(), h->value().c_str());
166 }
167
168 int params = request->params();
169 for (int i = 0; i < params; i++) {
170 AsyncWebParameter* p = request->getParam(i);
171 if (p->isFile()) {
172 debugD("_FILE[%s]: %s, size: %u\n", p->name().c_str(), p->value().c_str(),
173 p->size());
174 } else if (p->isPost()) {
175 debugD("_POST[%s]: %s\n", p->name().c_str(), p->value().c_str());
176 } else {
177 debugD("_GET[%s]: %s\n", p->name().c_str(), p->value().c_str());
178 }
179 }
180
181 request->send(404);
182}
