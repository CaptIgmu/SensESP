186void WSClient::on_receive_put(DynamicJsonDocument& message) {
187 // Process PUT requests...
188 JsonArray puts = message["put"];
189 size_t responseCount = 0;
190 for (size_t i = 0; i < puts.size(); i++) {
191 JsonObject put = puts[i];
192 const char* path = put["path"];
193 String strVal = put["value"].as<String>();
194 debugD("Received PUT request for path %s (value %s)\n", path,
195 strVal.c_str());
196 const std::vector<SKPutListener*>& listeners =
197 SKPutListener::get_listeners();
198 for (size_t i = 0; i < listeners.size(); i++) {
199 SKPutListener* listener = listeners[i];
200 if (listener->get_sk_path().equals(path)) {
201 listener->parse_value(put);
202 responseCount++;
203 }
204 }
205
206 // Send back a request reasponse...
207 DynamicJsonDocument putResponse(512);
208 putResponse["requestId"] = message["requestId"];
209 if (responseCount == puts.size()) {
210 // We found a response for every PUT request
211 putResponse["state"] = "COMPLETED";
212 putResponse["statusCode"] = 200;
213 } else {
214 // One or more requests did not have a matching path
215 putResponse["state"] = "FAILED";
216 putResponse["statusCode"] = 405;
217 }
218 String responseTxt;
219 serializeJson(putResponse, responseTxt);
220 debugD("Replying to PUT request with %s", responseTxt.c_str());
221 this->client_.sendTXT(responseTxt);
222 }

245void WSClient::connect() {
246 debugI("WSClient websocket connect attempt (state=%d)",
247 connection_state_.get());
248
249 if (connection_state_ != WSConnectionState::kWSDisconnected) {
250 return;
251 }
252
253 if (!WiFi.isConnected()) {
254 debugI(
255 "WiFi is disconnected. SignalK client connection will connect when "
256 "WiFi is connected.");
257 return;
258 }
259
260 debugD("Initiating websocket connection with server...");
261
262 connection_state_ = WSConnectionState::kWSAuthorizing;
263 String server_address = this->server_address_;
264 uint16_t server_port = this->server_port_;
265
266 if (this->server_address_.isEmpty()) {
267 if (!get_mdns_service(server_address, server_port)) {
268 debugE("No Signal K server found in network when using mDNS service!");
269 } else {
270 debugI("Signal K server has been found at address %s:%d by mDNS.",
271 server_address.c_str(), server_port);
272 }
273 }
274
275 if (!server_address.isEmpty() && server_port > 0) {
276 debugD("Websocket is connecting to Signal K server on address %s:%d",
277 server_address.c_str(), server_port);
278 } else {
279 // host and port not defined - wait for mDNS
280 connection_state_ = WSConnectionState::kWSDisconnected;
281 return;
282 }
283
284 if (this->polling_href_ != "") {
285 // existing pending request
286 this->poll_access_request(server_address, server_port, this->polling_href_);
287 return;
288 }
289
290 if (this->auth_token_ == NULL_AUTH_TOKEN) {
291 // initiate HTTP authentication
292 debugD("No prior authorization token present.");
293 this->send_access_request(server_address, server_port);
294 return;
295 }
296
297 // Test the validity of the authorization token
298 this->test_token(server_address, server_port);
299}

342void WSClient::send_access_request(const String server_address,
343 const uint16_t server_port) {
344 debugD("Preparing a new access request");
345 if (client_id_ == "") {
346 // generate a client ID
347 client_id_ = generate_uuid4();
348 save_configuration();
349 }
350
351 // create a new access request
352 DynamicJsonDocument doc(1024);
353 doc["clientId"] = client_id_;
354 doc["description"] =
355 String("SensESP device: ") + SensESPBaseApp::get_hostname();
356 doc["permissions"] = kRequestPermission;
357 String json_req = "";
358 serializeJson(doc, json_req);
359
360 HTTPClient http;
361
362 String url = String("http://") + server_address + ":" + server_port +
363 "/signalk/v1/access/requests";
364 http.begin(wifi_client_, url);
365 http.addHeader("Content-Type", "application/json");
366 int httpCode = http.POST(json_req);
367 String payload = http.getString();
368 http.end();
369
370 // if we get a response we can't handle, try to reconnect later
371 if (httpCode != 202) {
372 debugW("Can't handle response %d to access request.", httpCode);
373 debugD("%s", payload.c_str());
374 connection_state_ = WSConnectionState::kWSDisconnected;
375 client_id_ = "";
376 return;
377 }
378
379 // http status code 202
380
381 deserializeJson(doc, payload.c_str());
382 String state = doc["state"];
383
384 if (state != "PENDING") {
385 debugW("Got unknown state: %s", state.c_str());
386 connection_state_ = WSConnectionState::kWSDisconnected;
387 client_id_ = "";
388 return;
389 }
390
391 String href = doc["href"];
392 polling_href_ = href;
393 save_configuration();
394
395 debugD("Polling %s in 5 seconds", polling_href_.c_str());
396 ReactESP::app->onDelay(5000, [this, server_address, server_port, href]() {
397 this->poll_access_request(server_address, server_port, this->polling_href_);
398 });
399}

401void WSClient::poll_access_request(const String server_address,
402 const uint16_t server_port,
403 const String href) {
404 debugD("Polling SK Server for authentication token");
405
406 HTTPClient http;
407
408 String url = String("http://") + server_address + ":" + server_port + href;
409 http.begin(wifi_client_, url);
410 int httpCode = http.GET();
411 if (httpCode == 200 or httpCode == 202) {
412 String payload = http.getString();
413 http.end();
414 DynamicJsonDocument doc(1024);
415 auto error = deserializeJson(doc, payload.c_str());
416 if (error) {
417 debugW("WARNING: Could not deserialize http payload.");
418 debugW("DeserializationError: %s", error.c_str());
419 return; // TODO: return at this point, or keep going?
420 }
421 String state = doc["state"];
422 debugD("%s", state.c_str());
423 if (state == "PENDING") {
424 ReactESP::app->onDelay(5000, [this, server_address, server_port, href]() {
425 this->poll_access_request(server_address, server_port, href);
426 });
427 return;
428 } else if (state == "COMPLETED") {
429 JsonObject access_req = doc["accessRequest"];
430 String permission =
431 access_req["permission"]; // TODO: like this in ArdJson 6? String
432 // permission =
433 // resp["accessRequest"]["permission"];
434 polling_href_ = "";
435 save_configuration();
436
437 if (permission == "DENIED") {
438 debugW("Permission denied");
439 connection_state_ = WSConnectionState::kWSDisconnected;
440 return;
441 } else if (permission == "APPROVED") {
442 debugI("Permission granted");
443 String token = access_req["token"];
444 auth_token_ = token;
445 save_configuration();
446 ReactESP::app->onDelay(0, [this, server_address, server_port]() {
447 this->connect_ws(server_address, server_port);
448 });
449 return;
450 }
451 }
452 } else {
453 http.end();
454 if (httpCode == 500) {
455 // this is probably the server barfing due to
456 // us polling a non-existing request. Just
457 // delete the polling href.
458 debugD("Got 500, probably a non-existing request.");
459 polling_href_ = "";
460 save_configuration();
461 connection_state_ = WSConnectionState::kWSDisconnected;
462 return;
463 }
464 // any other HTTP status code
465 debugW("Can't handle response %d to pending access request.\n", httpCode);
466 connection_state_ = WSConnectionState::kWSDisconnected;
467 return;
468 }
